	id = EDG_SWIEvents.2
	is_triggered_only = yes

	desc = EDG_SWIEvents.2.desc
	picture = DIPLOMACY_eventPicture

	trigger = {
		is_subject = no
	}

	option = {
		name = EDG_SWIEvents.2.aa
		1768 = {
			if = {
				limit = { 
					NOT = {
						is_core = TIR
					}
				}
				add_core = TIR
			}
			owner = {
				release = TIR
				root = {
					declare_war_with_cb = {
						casus_belli = cb_annex
						who = prev
					}
				}
			}
		}
		if = {
			limit = { 
				has_government_attribute = EDG_enable_freedom_allian
			}
			EDG_SWI_create_swi_state = {
				who = TIR
			}
		}
		else = {
			create_subject = {
				subject_type = vassal
				subject = TIR
			}
		}
	}
	option = {
		name = EDG_SWIEvents.2.bb
		add_prestige = -10
		custom_tooltip = EDG_SWIEvents.2.bb.tt
		hidden_effect = {
			set_country_flag = EDG_SWI_declar_war_with_1768_flag
		}
	}
}
