country_event = {
  id = EDG_PER_disaster_throne_war.1
  title = "EDG_PER_disaster_throne_war.1.t"
  desc = "EDG_PER_disaster_throne_war.1.d"
  picture = CIVIL_WAR_eventPicture
  is_triggered_only = yes
  major = yes
  immediate = {
    hidden_effect = {
		log = "[Root.GetName] has EDG_PER_disaster_throne_war Disaster."
		set_country_flag = EDG_PER_disaster_throne_war_flag
	}
    hidden_effect = {
		add_stability = -6
		
		set_variable = {
			which = EDG_PER_disaster_throne_war_power
			value = 0
		}
		add_country_modifier = {
			name = EDG_PER_disaster_throne_war_reform_decision_enable_modifier
			duration = -1
			hidden = yes
		}
    }
  }
  option = {
    name = "EDG_PER_disaster_throne_war.1.a"
	add_prestige = -10
	
    if = {
      limit = {
        any_owned_province = {
          development = 5
          highest_supply_limit_in_area = yes
          is_in_capital_area = yes
          fort_level = 1
          is_capital = no
        }
      }
      random_owned_province = {
		add_local_autonomy = 25
		add_devastation = 20
		spawn_rebels = {
			size = 3
			win = no
			unrest = 20
		  }
      }
      random_owned_province = {
		add_local_autonomy = 25
		add_devastation = 20
		spawn_rebels = {
			type = noble_rebels
			size = 3
			win = no
			unrest = 20
		  }
      }
      random_owned_province = {
		add_local_autonomy = 25
		add_devastation = 20
		spawn_rebels = {
			type = noble_rebels
			size = 3
			win = no
			unrest = 20
		  }
      }
    }
    else = {
      random_owned_province = {
        limit = {
          highest_supply_limit_in_area = yes
          is_capital = no
        }
		
		add_local_autonomy = 25
		add_devastation = 20
		spawn_rebels = {
			size = 3
			win = no
			unrest = 20
		  }
      }
    }
    if = {
      limit = {
        NOT = {
          has_dlc = "Emperor"
        }
      }
      random_owned_province = {
        limit = {
          is_capital = yes
        }
		add_local_autonomy = 25
		add_devastation = 20
		spawn_rebels = {
			type = noble_rebels
			size = 3
			win = no
			unrest = 20
		  }
      }
    }
	tooltip = { add_stability = -6 }
    
  }
  
}
