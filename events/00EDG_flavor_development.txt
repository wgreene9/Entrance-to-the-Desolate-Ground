country_event = {#移民的到来
	id = EDG_flavor_development.0
	title = "EDG_flavor_development.0.tt"
	desc = "EDG_flavor_development.0.dd"
	picture = FAMINE_eventPicture
	goto = EDG_flavor_development_0_province
	
	trigger = {
		any_owned_province = { is_city = yes NOT = { development = 20 } }
	}
	
	is_triggered_only = yes


	immediate = {
		hidden_effect = {
			random_owned_province = {
				limit = {
					is_city = yes
					NOT = { development = 20 }
				}
				save_event_target_as = EDG_flavor_development_0_province
			}
		}
	}
	
	option = {#欢迎他们的到来！
		name = "EDG_flavor_development.0.a"
		ai_chance = {
		  factor = 2
		}
		add_adm_power = -25
		add_dip_power = -25
		event_target:EDG_flavor_development_0_province = {
			random_list = {
				10 = { add_base_tax = 1 }
				10 = { add_base_production = 1 }
				10 = { add_base_manpower = 1 }
			}
		}
	}
	option = {#我们才不接受他们。
		name = "EDG_flavor_development.0.b"
		ai_chance = {
		  factor = 1
		}
		add_prestige = -5
	}
}
