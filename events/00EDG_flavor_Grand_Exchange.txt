	title = "EDG_flavor_Grand_Exchange.0.title"
	desc = "EDG_flavor_Grand_Exchange.0.desc"
	picture = IMPORTANT_STATUE_eventPicture
	major = yes
	fire_only_once = yes
	is_triggered_only = yes
	
	trigger = {
		NOT = { is_year = 1600 }
		technology_group = western

			num_of_colonists = 1
			capital_scope = { region = iberia_region }
			capital_scope = { region = british_isles_region }
			capital_scope = { region = france_region }
		}
		OR = {
			is_year = 1521
			calc_true_if = {
				all_province = {
					OR = {
						superregion = north_america_superregion
						superregion = central_america_superregion
						superregion = andes_superregion
						superregion = south_america_superregion
					}
					OR = {
						is_colony = yes
						owner = { technology_group = western }
						owner = { is_colonial_nation = yes }
						owner = { capital_scope = { superregion = europe_superregion } }
					}
				}
				amount = 25
			}
		}
	}
	

	mean_time_to_happen = {
		
	}

	immediate = {
		
	}
	
	option = {#这是惩罚还是机遇？

		  factor = 2
		}
		set_global_flag = EDG_Grand_Exchange_global_flag

	}
