	picture = SEJM_eventPicture

	trigger = {
		culture_group = east_slavic
		government = monarchy
		is_female = no
		has_ruler_modifier = EDG_rus_genaral_rurikovich_modifier
		NOT = {
			is_monarch_leader = yes
		}
		NOT = {
			has_ruler_modifier = EDG_rus_afraid_ruler
		}
		NOT = {
			has_ruler_modifier = EDG_rus_refuse_ruler
		}
		is_at_war = yes
	}

	option = { 
		name = "EDG_BELA_rus_genaral_events.2A"
		add_ruler_modifier = {
			name = EDG_rus_brief_ruler
		}
		trigger = {
			NOT = {
				ruler_has_personality = craven_personality
			}
		}
		if = {
			limit = {
				army_tradition = 75
			}
			convert_ruler_to_general = 100
		}
		else_if = {
			limit = {
				army_tradition = 50
			}
			convert_ruler_to_general = 75
		}
		else_if = {
			limit = {
				army_tradition = 25
