EDG_DMR_cultural_assimilation_effect = {
	random_owned_province = {
		limit = {
			EDG_DMR_province_can_be_assimilated_trigger = yes
		}
		save_event_target_as = EDG_DMR_cultural_assimilation_province_target_1_$id$
		ROOT = {
			set_country_flag = EDG_DMR_cultural_assimilation_flag_1_$id$
			export_to_variable = {
				variable_name = EDG_DMR_assimilated_culture_var
				value = culture
				who = PREV
			}
		}
		EDG_DMR_change_culture_effect = yes
		if = {
			limit = {
				ROOT = {
					has_country_flag = EDG_DMR_mission_assimilate_2_flag
					accepted_culture = variable:From:EDG_DMR_assimilated_culture_var
				}
			}
			ROOT = {
				random_owned_province = {
					limit = {
						EDG_DMR_province_can_be_assimilated_trigger = yes
						culture = variable:From:EDG_DMR_assimilated_culture_var
					}
					save_event_target_as = EDG_DMR_cultural_assimilation_province_target_2_$id$
					ROOT = {
						set_country_flag = EDG_DMR_cultural_assimilation_flag_2_$id$
					}
					EDG_DMR_change_culture_effect = yes
				}
			}
		}
	}
}

EDG_DMR_change_culture_effect = {
	if = {
		limit = {
			ROOT = {
				culture_group = east_slavic
			}
		}
		change_culture = EDG_rom_rus_culture
	}
	else = {
		change_culture = EDG_rom_culture
	}
}
