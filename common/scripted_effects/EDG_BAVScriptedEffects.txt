EDG_BAV_startup_effect = {
	hidden_effect = {
		add_country_modifier = {
			name = EDG_BAV_bavarian_ambition_modifier
			duration = -1
		}
		if = {
			limit = {
				ai = no
				WUR = {
					ai = yes
				}
			}
			WUR = {
				set_ai_attitude = {
					attitude = attitude_hostile
					who = ROOT
				}
			}
			every_country = {
				limit = {
					OR = {
						tag = BOH
						tag = H02
						tag = HUN
					}
				}
				set_ai_attitude = {
					attitude = attitude_friendly
					who = WUR
				}
				remove_historical_rival = WUR
				add_historical_friend = WUR
				add_opinion = {
					who = WUR
					modifier = EDG_BAV_emperor_opinion_modifier
				}
				WUR = {
					set_ai_attitude = {
						attitude = attitude_friendly
						who = PREV
					}
					remove_historical_rival = PREV
					add_historical_friend = PREV
					add_opinion = {
						who = PREV
						modifier = EDG_BAV_emperor_opinion_modifier
					}
				}
			}
		}
	}
}

EDG_BAV_create_subject_effect = {
	create_subject = {
		who = $who$
		subject_type = EDG_BAV_Wittelsbach_Territories
	}
	hidden_effect = {
		$who$ = {
			if = {
				limit = {
					NOT = {
						government = monarchy
					}
				}
				change_government = monarchy
			}
			add_government_reform = EDG_BAV_wittelsbach_subject
		}
	}
}

EDG_BAV_add_permanent_claims_effect = {
	custom_tooltip = EDG_BAV_add_permanent_claims_effect_tt
	hidden_effect = {
		every_neighbor_province = {
			limit = {
				EDG_BAV_is_german_province_trigger = yes
				NOT = {
					country_or_non_sovereign_subject_holds = ROOT
				}
			}
			area = {
				add_permanent_claim = ROOT
			}
		}
		every_subject_country = {
			every_neighbor_province = {
				limit = {
					EDG_BAV_is_german_province_trigger = yes
					NOT = {
						country_or_non_sovereign_subject_holds = ROOT
					}
				}
				area = {
					add_permanent_claim = ROOT
				}
			}
		}
	}
}
