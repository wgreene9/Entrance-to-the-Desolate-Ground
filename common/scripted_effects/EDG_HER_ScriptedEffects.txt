EDG_HER_republic_level_up = {
	if = {
		limit = {
			ruler_has_personality = EDG_personality_republican_knowledge_1
		}
		remove_ruler_personality = EDG_personality_republican_knowledge_1
		add_ruler_personality = EDG_personality_republican_knowledge_2
	}
	else_if = {
		limit = {
			ruler_has_personality = EDG_personality_republican_knowledge_2
		}
		remove_ruler_personality = EDG_personality_republican_knowledge_2
		add_ruler_personality = EDG_personality_republican_knowledge_3
	}
	else_if = {
		limit = {
			ruler_has_personality = EDG_personality_republican_knowledge_3
		}
		remove_ruler_personality = EDG_personality_republican_knowledge_3
		add_ruler_personality = EDG_personality_republican_knowledge_4
	}
	else_if = {
		limit = {
			ruler_has_personality = EDG_personality_republican_knowledge_4
		}
		remove_ruler_personality = EDG_personality_republican_knowledge_4
		add_ruler_personality = EDG_personality_republican_knowledge_5
	}
	else_if = {
		limit = {
			ruler_has_personality = EDG_personality_republican_knowledge_5
		}
		add_splendor = 100
	}
	else = {
		add_ruler_personality = EDG_personality_republican_knowledge_1
	}
}

EDG_HER_be_the_one_level_up = {
	if = {
		limit = {
			has_country_modifier = EDG_HER_be_the_one_level_up_1_modifier
		}
		remove_country_modifier = EDG_HER_be_the_one_level_up_1_modifier
		add_country_modifier = {
			name = EDG_HER_be_the_one_level_up_2_modifier
			duration = -1
		}
	}
	else_if = {
		limit = {
			has_country_modifier = EDG_HER_be_the_one_level_up_2_modifier
		}
		remove_country_modifier = EDG_HER_be_the_one_level_up_2_modifier
		add_country_modifier = {
			name = EDG_HER_be_the_one_level_up_3_modifier
			duration = -1
		}
	}
	else_if = {
		limit = {
			has_country_modifier = EDG_HER_be_the_one_level_up_3_modifier
		}
		remove_country_modifier = EDG_HER_be_the_one_level_up_3_modifier
		add_country_modifier = {
			name = EDG_HER_be_the_one_level_up_4_modifier
			duration = -1
		}
	}
	else_if = {
		limit = {
			has_country_modifier = EDG_HER_be_the_one_level_up_4_modifier
		}
		remove_country_modifier = EDG_HER_be_the_one_level_up_4_modifier
		add_country_modifier = {
			name = EDG_HER_be_the_one_level_up_5_modifier
			duration = -1
		}
	}
	else_if = {
		limit = {
			has_country_modifier = EDG_HER_be_the_one_level_up_5_modifier
		}
		add_prestige_or_monarch_power = {
			amount = 10
		}
	}
	else = {
		add_country_modifier = {
			name = EDG_HER_be_the_one_level_up_1_modifier
			duration = -1
		}
	}
}

EDG_CWN_gangkoucheng_effect = {
	#给省份上港口城
	if = {
		limit = {
			num_of_owned_provinces_with = {
				has_port = yes
				NOT = {
					has_province_modifier = EDG_HER_2_2_modifier
				}
				value = 1
			}
		}
		random_owned_province = {
			limit = {
				has_port = yes
				NOT = {
					has_province_modifier = EDG_HER_2_2_modifier
				}
			}
			add_province_modifier = {
				name = EDG_HER_2_2_modifier
				duration = -1
			}
		}
	}
	else = {
		add_sailors = 1
	}
}

EDG_CWN_daxue_effect = {
	if = {
		limit = {
			num_of_owned_provinces_with = {
				development = 30
				NOT = {
					has_building = university
				}
				value = 1
			}
		}
		random_owned_province = {
			limit = {
				development = 30
				NOT = {
					has_building = university
				}
			}
			add_building = university
		}
	}
	else = {
		add_treasury = 10
	}
}
