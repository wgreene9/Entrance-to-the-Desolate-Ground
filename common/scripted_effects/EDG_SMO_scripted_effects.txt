EDG_SMO_revolutionary_tendency_back_center_common_effect = {
	custom_tooltip = EDG_SMO_revolutionary_tendency_back_center_common.tt
	hidden_effect = {
		if = {
			limit = { 
				has_government_power = {
					mechanic_type = EDG_SMO_revolutionary_tendency_mechanic
					power_type = EDG_SMO_revolutionary_tendency
					value = 20
				}
			}
			add_government_power = {
				mechanic_type = EDG_SMO_revolutionary_tendency_mechanic
				power_type = EDG_SMO_revolutionary_tendency
				value = -20
			}
		}
		else_if = {
			limit = { 
				NOT = {
					has_government_power = {
						mechanic_type = EDG_SMO_revolutionary_tendency_mechanic
						power_type = EDG_SMO_revolutionary_tendency
						value = -20
					}
				}
			}
			add_government_power = {
				mechanic_type = EDG_SMO_revolutionary_tendency_mechanic
				power_type = EDG_SMO_revolutionary_tendency
				value = 20
			}
		}
		else = {
			set_government_power = {
				mechanic_type = EDG_SMO_revolutionary_tendency_mechanic
				power_type = EDG_SMO_revolutionary_tendency
				value = 0
			}
		}
	}
}

	]

			mechanic_type = EDG_SMO_estate_influence_mechanic

		}
	]
}
