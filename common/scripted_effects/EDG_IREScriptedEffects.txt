#limit 省份条件 /optional
#owner 所有者	/necessary
EDG_remove_core_effect = {
	hidden_effect = {
		every_owned_province = {
			limit = {
				[[limit]
					$limit$
				]
			}
			every_core_country = {
				limit = {
					NOT = {
						tag = $owner$
					}
				}
				remove_core = prev
			}
		}
	}
}
EDG_remove_religious_zeal_at_conv_effect = {
	custom_tooltip = EDG_remove_religious_zeal_at_conv_effect.tt
	hidden_effect = {
		every_owned_province = {
			limit = {
				[[limit]
					$limit$
				]
				has_province_modifier = religious_zeal_at_conv
				NOT = {
					religion = root
				}
			}
			remove_province_modifier = religious_zeal_at_conv
		}
	}
}
