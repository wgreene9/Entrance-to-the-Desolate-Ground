EDG_add_claim = {
	$area$ = {
		limit = {
			NOT = { is_core = root }
			NOT = { is_permanent_claim = root }
			# NOT = { owner = { is_subject_of = root } }
		}
		add_permanent_claim = root
	}
}
EDG_SWI_create_freedom_allian = {
	create_subject = {
		subject_type = EDG_SWI_free_alliance
		subject = $who$
	}
	$who$ = {
		change_government = republic
		add_government_reform = EDG_SWI_freedom_allian_subject
		add_historical_friend = prev
	}
	add_historical_friend = $who$
}
