effect_and_subtract_var_if_var_greater = {
    if = {
        limit = {
            check_variable = {
                which = $VAR$
                value = $VALUE$
            }
        }
        subtract_variable = {
            which = $VAR$
            value = $VALUE$
        }
        $EFFECTNAME$ = $VALUE$
    }
}
