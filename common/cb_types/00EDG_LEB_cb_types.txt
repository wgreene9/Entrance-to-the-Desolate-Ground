00EDG_LEB_cb_spread_islam = {#传播伊斯兰教
	valid_for_subject = yes
	holy_war = yes
	
	prerequisites_self = {
		has_country_flag = EDG_LEB_cb_flag
	}
	prerequisites = {
		FROM = {
			NOT = { religion_group = muslim }
		}
		OR = {
			AND = {
				has_country_flag = EDG_LEB_M_CB_I_flag				
				FROM = {
					capital_scope = {
						region = italy_region
					}
				}
			}
			AND = {
				has_country_flag = EDG_LEB_M_CB_W_flag	
				FROM = {
					capital_scope = {
						OR = {
							region = france_region
							region = iberia_region
						}
					}
				}
			}
			AND = {
				has_country_flag = ARB_cb_on_fetishists_in_south_africa_flag
				FROM = {
					capital_scope = {
						superregion = southern_africa_superregion
					}
				}
			}
			AND = {
				
				FROM = {
					capital_scope = { region = balkan_region }
				}
			}
		}
	}
	
	war_goal = ARB_spread_islam_wargoal
}
