estate_edg_artists_agenda_1 = {
	can_select = {
		has_government_mechanic = EDG_BAV_government_ability
	}
	selection_weight = {
		factor = 10
	}
	pre_effect = {
		set_variable = {
			which = estate_edg_artists_agenda_1_variable
			which = EDG_BAV_government_power_interaction_1_variable
		}
	}
	task_requirements = {
		if = {
			limit = {
				has_estate_privilege = estate_edg_artists_privilege_5
			}
			always = yes
		}
		else = {
			custom_trigger_tooltip = {
				tooltip = estate_edg_artists_agenda_1_tt
				NOT = {
					is_variable_equal = {
						which = estate_edg_artists_agenda_1_variable
						which = EDG_BAV_government_power_interaction_1_variable
					}
				}
			}
		}
	}
	task_completed_effect = {
		add_estate_loyalty = {
			estate = estate_edg_artists
			loyalty = 20
		}
		add_government_power = {
			mechanic_type = EDG_BAV_government_ability
			power_type = EDG_BAV_government_power
			value = 2
		}
	}
	failing_effect = {
		add_estate_loyalty_modifier = {
			estate = estate_edg_artists
			desc = EST_VAL_AGENDA_DENIED
			loyalty = -5
			duration = 7300
		}
	}
}

estate_edg_artists_agenda_5 = {
	can_select = {
		any_subject_country = {
			is_subject_of_type = EDG_BAV_Wittelsbach_Territories
		}
	}
	selection_weight = {
		factor = 10
	}
	pre_effect = {
		random_province = {
			limit = {
				owner = {
					is_subject_of_type_with_overlord = {
						who = ROOT
						type = EDG_BAV_Wittelsbach_Territories
					}
				}
			}
			save_event_target_as = agenda_province
		}
		event_target:agenda_province = {
			set_variable = {
				which = province_adm_var
				value = 0
			}
			export_to_variable = {
				which = province_adm_var
				value = trigger_value:base_tax
			}
			change_variable = {
				which = province_adm_var
				value = 2
			}
		}
	}
	provinces_to_highlight = {
		province_id = event_target:agenda_province
	}
	task_requirements = {
		if = {
			limit = {
				has_estate_privilege = estate_edg_artists_privilege_5
			}
			always = yes
		}
		else = {
			event_target:agenda_province = {
				base_tax = "province_adm_var"
			}
		}
	}
	task_completed_effect = {
		add_estate_loyalty = {
			estate = estate_edg_artists
			loyalty = 20
		}
		event_target:agenda_province = {
			owner = {
				add_liberty_desire = -25
			}
		}
	}
	fail_if = {
		event_target:agenda_province = {
			NOT = {
				country_or_non_sovereign_subject_holds = ROOT
			}
		}
	}
	failing_effect = {
		add_estate_loyalty_modifier = {
			estate = estate_edg_artists
			desc = EST_VAL_AGENDA_DENIED
			loyalty = -5
			duration = 7300
		}
	}
}
