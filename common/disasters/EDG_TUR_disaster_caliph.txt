EDG_TUR_disaster_caliph = {#哈里发地位受质疑
	ended_by_country_breaking_to_rebels = no
	potential = {
		NOT = { has_country_flag = EDG_TUR_disaster_caliph_end_flag }
		tag = TUR
		is_free_or_tributary_trigger = yes
		uses_doom = no
		normal_or_historical_nations = yes
		is_revolutionary = no
	}
	
	
	can_start = {
		has_any_disaster = no
		NOT = { current_age = age_of_discovery }
		has_country_modifier = EDG_Usurping_the_sunni_Caliphate_modifier
		religion = sunni
	}
	
	can_stop = {
		has_any_disaster = yes
		not = { religion = sunni }
		not = { has_country_modifier = EDG_Usurping_the_sunni_Caliphate_modifier }
	}
	
	
	progress = {
		# modifier = {
		# 	factor = 100
		# 	always = yes
		# }
		modifier = {
			factor = 0.5
			NOT = { adm = 6 }
		}
		modifier = {
			factor = 0.5
			NOT = { dip = 6 }
		}
		modifier = {
			factor = 0.5
			NOT = { mil = 6 }
		}
		modifier = {
			factor = 0.5
			NOT = { prestige = 0 }
		}
		modifier = {
			factor = 0.5
			war_exhaustion = 1
		} 
		modifier = {
			factor = 0.75
			num_of_loans = 10
			NOT = { num_of_loans = 25 }
		}
		modifier = {
			factor = 1.5
			num_of_loans = 25
		}
		modifier = {
			factor = 1
			capital_scope = { controlled_by = REB }
		}
		modifier = {
			factor = 1
			is_bankrupt = yes
		}
	}
	
	can_end = {
		OR = {
			custom_trigger_tooltip = {
				tooltip = disaster_active_for_20_years_tooltip
				had_country_flag = {
					flag = EDG_TUR_disaster_caliph_flag
					days = 7300
				}
			}
			AND = {
				stability = 2
				NOT = { num_of_rebel_controlled_provinces = 1 }
				NOT = { num_of_rebel_armies = 1 }
				has_country_flag = EDG_TUR_disaster_caliph_11_a_flag
				has_country_flag = EDG_TUR_disaster_caliph_11_b_flag
				has_country_flag = EDG_TUR_disaster_caliph_11_c_flag
				has_country_flag = EDG_TUR_disaster_caliph_11_d_flag
				has_country_flag = EDG_TUR_disaster_caliph_11_e_flag
			}
		}
	}
	

	modifier = {
		tolerance_own = -5
		institution_spread_from_true_faith = -0.33
		manpower_in_true_faith_provinces = -0.33
		monthly_ottoman_decadence = 0.1
		stability_cost_modifier = 0.25
	}	
	
	
	on_monthly = {
		events = {
			EDG_TUR_disaster_caliph.10
		}
		random_events = { 
			1000 = 0
			100 = EDG_TUR_disaster_caliph.3
			100 = EDG_TUR_disaster_caliph.4
			100 = EDG_TUR_disaster_caliph.5
			100 = EDG_TUR_disaster_caliph.6
			100 = EDG_TUR_disaster_caliph.7
			100 = EDG_TUR_disaster_caliph.8
			100 = EDG_TUR_disaster_caliph.9
		}
	}
	
	
	on_start = EDG_TUR_disaster_caliph.1 #General Estates / Storming of the Bastille
	on_end = EDG_TUR_disaster_caliph.2 #Long live the Republic/Monarchy
	
}
