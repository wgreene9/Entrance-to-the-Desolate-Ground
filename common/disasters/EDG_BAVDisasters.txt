EDG_BAV_religion_disaster = {
	#安魂曲
	potential = {
		EDG_BAV_is_bavarian_country_trigger = yes
		NOT = {
			has_country_flag = EDG_BAV_religion_disaster_ended_flag
		}
	}
	can_start = {
		custom_trigger_tooltip = {
			tooltip = EDG_BAV_religion_disaster_can_start_tt
			has_country_flag = EDG_BAV_religion_disaster_can_start_flag
		}
		custom_trigger_tooltip = {
			tooltip = EDG_BAV_religion_disaster_tt
			always = yes
		}
	}
	can_stop = {
		always = no
	}
	can_end = {
		if = {
			limit = {
				ai = no
			}
			is_at_war = no
			stability = 0
			NOT = {
				num_of_rebel_armies = 1
			}
			NOT = {
				num_of_rebel_controlled_provinces = 1
			}
			all_subject_country = {
				NOT = {
					num_of_rebel_armies = 1
				}
				NOT = {
					num_of_rebel_controlled_provinces = 1
				}
			}
			check_variable = {
				which = EDG_BAV_religion_disaster_variable
				value = 15
			}
			custom_trigger_tooltip = {
				tooltip = EDG_BAV_religion_disaster_can_end_tt
				has_country_flag = EDG_BAV_religion_disaster_can_end_flag
			}
		}
		else = {
			ai = yes
		}
	}
	progress = {
		modifier = {
			factor = 10
			always = yes
		}
	}
	modifier = {
		tolerance_heretic = -10
		tolerance_own = -10
		stability_cost_modifier = 0.5
		global_heretic_missionary_strength = -1
		missionary_maintenance_cost = 10
		religious_unity = -1
		papal_influence = -100
	}
	on_start = EDG_BAV_event.27
	on_end = EDG_BAV_event.28
	on_monthly = {
		random_events = {
			850 = 0
			30 = EDG_BAV_event.35
			30 = EDG_BAV_event.36
			30 = EDG_BAV_event.37
			30 = EDG_BAV_event.38
			30 = EDG_BAV_event.39
		}
	}
}
