on_startup = {
	if = {
		limit = {
			tag = MEN
		}
		country_event = {
			id = EDG_flavorMEN.100
			days = 1
		}
	}
}

on_province_owner_change = {
	if = {
		limit = {
			owner = {
				has_reform = EDG_MEN_hella_noble_reform
			}
			culture_group = byzantine
		}
		add_province_modifier = {
			name = EDG_MEN_culture_modifier
			duration = -1
		}
	}
}

on_war_won = {
	if = {
		limit = {
			has_country_flag = EDG_MEN_3_16_oa_flag
		}
		ROOT = {
			add_splendor = 10
		}
	}
}

on_siege_won_country = {
	if = {
		limit = {
			from = {
				# owned_by = MOE
				is_capital_of = MOE
				controlled_by = root
			}
			root = {
				OR = {
					tag = MEN
					is_in_war = {
						defenders = MEN
						attackers = MOE
					}
					is_in_war = {
						defenders = MOE
						attackers = MEN
					}						
				}
			}
		}
		MEN = {
			country_event = {
				id = EDG_flavorMEN.102
			}
		}
	}
}
