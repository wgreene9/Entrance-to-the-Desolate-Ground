on_culture_promoted = {
	if = {
		limit = {
			has_country_flag = EDG_H06_get_power_when_promot_culture_flag
		}
		change_government_reform_progress = 20
	}
	if = {
		limit = { 
			has_country_flag = EDG_H06_enable_promoted_culture_for_change_flag
		}
		every_owned_province = {
			limit = {
				NOT = {
					culture_group = root
				}
				has_owner_accepted_culture = yes
			}
			add_province_modifier = {
				name = EDG_H06_assimilate_culture
				duration = -1
			}
		}
	}
}
on_culture_demoted = {
	if = {
		limit = {
			has_country_flag = EDG_H06_get_power_when_promot_culture_flag
		}
		change_government_reform_progress = -20
	}
	if = {
		limit = { 
			has_country_flag = EDG_H06_enable_promoted_culture_for_change_flag
		}
		every_owned_province = {
			limit = {
				has_province_modifier = EDG_H06_assimilate_culture
				has_owner_accepted_culture = no
			}
			remove_province_modifier = EDG_H06_assimilate_culture
		}
	}
}
on_province_culture_converted = {
	if = {
		limit = { 
			has_province_modifier = EDG_H06_assimilate_culture
		}
		remove_province_modifier = EDG_H06_assimilate_culture
	}
	if = {
		limit = { 
			owner = {
				has_country_flag = EDG_H06_mission_2_23_flag
			}
		}
		owner = {
			add_adm_power = 10
		}
	}
}
on_province_culture_changed = {
	if = {
		limit = { 
			has_province_modifier = EDG_H06_assimilate_culture
		}
		remove_province_modifier = EDG_H06_assimilate_culture
	}
	if = {
		limit = { 
			owner = {
				has_country_flag = EDG_H06_mission_2_23_flag
			}
		}
		owner = {
			add_adm_power = 10
		}
	}
}
on_diplomatic_annex = {
	if = {
		limit = {
			has_country_flag = EDG_H06_add_power_on_diplomatic_annex_flag
		}
		add_adm_power = 50
		add_dip_power = 50
		add_mil_power = 50
	}
	if = {
		limit = { 
			has_country_flag = EDG_H06_reduce_liberty_desire_flag
		}
		every_subject_country = {
			remove_opinion = {
				who = prev
				modifier = opinion_annex_vassal
			}
			add_liberty_desire = -10
		}
	}
	if = {
		limit = { 
			has_country_flag = EDG_H06_enable_get_land_on_diplomatic_annex_flag
		}
		change_estate_land_share = {
			estate = all
			share = -1
		}
	}
}
on_annexed = {
	if = {
		limit = {
			has_country_flag = EDG_H06_add_power_on_diplomatic_annex_flag
		}
		add_adm_power = 50
		add_dip_power = 50
		add_mil_power = 50
	}
	if = {
		limit = {
			has_country_flag = EDG_H06_enable_get_land_on_annex_flag
		}
		add_adm_power = 50
		add_dip_power = 50
		add_mil_power = 50
	}
}
