on_startup = {
	if = {
		limit = {
			tag = LEB
		}
		country_event = {
			id = EDG_LEB_flavor_event.1
			days = 1
		}
	}
}

# province
on_adm_development = {
	if = {
		limit = {
			has_climate = arid
			owner = {
				OR = {
					tag = LEB
					was_tag = LEB
				}
			}
		}
		owner = {
			change_variable = {
				which = num_of_times_developed_arid
				value = 1
			}
		}
	}
	if = {
		limit = {
			OR = {
				has_terrain = hills
				has_terrain = mountain
				has_terrain = highlands
			}
			owner = {
				OR = {
					tag = LEB
					was_tag = LEB
				}
			}
		}
		owner = {
			change_variable = {
				which = EDG_num_of_times_developed_mountain
				value = 1
			}
		}
	}
	if = {
		limit = {
			owner = {
				has_country_flag = EDG_LEB_4_16_flag
			}
			NOT = {
				religion = ibadi
			}
		}
		random = {
			change_religion = ibadi
			chance = 25
		}
	}
	if = {
		limit = {
			trade_goods = livestock
			OR = {
				region = arabia_region
				culture = bedouin_arabic
			}
			not = {
				has_province_modifier = EDG_LEB_fe_9_d_modifier
			}
			owner = {
				has_country_flag = EDG_LEB_2_18_flag
			}
		}
		random = {
			province_event = {
				id = EDG_LEB_flavor_event.9
			}
			chance = 25
		}
	}
	if = {
		limit = {
			owner = {
				has_country_flag = EDG_LEB_1_21_flag
			}
		}
		owner = {
			add_splendor = 1
		}
	}
	if = {
		limit = {
			OR = {
				has_terrain = coastal_desert
				has_terrain = desert
			}
			has_province_modifier = EDG_LEB_grass_block_modifier
			not = {
				has_province_modifier = EDG_LEB_fe_10_modifier
			}
			owner = {
				has_country_flag = EDG_LEB_3_24_flag
			}
		}
		random = {
			province_event = {
				id = EDG_LEB_flavor_event.10
			}
			chance = 25
		}
	}
}

on_province_owner_change = {
	if = {
		limit = {
			owner = {
				has_reform = EDG_LEB_dip_army
			}
			OR = {
				trade_goods = grain
				trade_goods = livestock
			}
		}
		add_province_modifier = {
			name = grain_bonus_prod
			duration = -1
			hidden = yes
		}
	}
}
