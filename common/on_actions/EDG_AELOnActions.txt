on_province_religion_converted = {
	if = {
		limit = { 
			religion = EDG_catharism
			exists = AEL
			AEL = { 
				has_government_mechanic = EDG_AEL_cathari_imperial_authority_mechanic
			}
		}
		AEL = {
			add_government_power = {
				mechanic_type = EDG_AEL_cathari_imperial_authority_mechanic
				power_type = EDG_AEL_cathari_imperial_authority
				value = 0.25
			}
		}
	}
}
on_startup = {
	if = {
		limit = {
			tag = AEL
		}
		country_event = {
			id = EDG_AELEvents.0
			days = 1825
			random = 1825
		}
	}
}
