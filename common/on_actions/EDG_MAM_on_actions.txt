on_province_religion_converted = {
	if = {
		limit = {
			owner = {
				has_country_flag = EDG_MAM_enable_convert_reduce_nationalism_flag
			}
		}
		add_nationalism = -10
		add_devastation = -10
	}
}

on_tech_taken = {
	if = {
		limit = {
			owner = {
				has_country_flag = EDG_MAM_tech_flag
			}
		}
		add_adm_power = 50 
		add_dip_power = 50
		add_mil_power = 50
		change_innovativeness = 5
	}
}
