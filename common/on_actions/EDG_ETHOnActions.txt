on_startup = {
	if = {
		limit = {
			tag = ETH
		}
		country_event = {
			id = EDG_ETHEvents.0
		}
	}

}
on_capital_moved = {
	events = {
		flavor_eth.100
	}
	if = {
		limit = {
			owner = {
				has_government_mechanic = EDG_ETH_solomonic_authority_mechanic
			}
		}
		owner = {
			add_government_power = {
				mechanic_type = EDG_ETH_solomonic_authority_mechanic
				power_type = EDG_ETH_solomonic_authority
				value = 2.5
			}
			change_variable = {
				which = EDG_ETH_moving_capital_count
				value = 1
			}
		}
		
	}
}
on_monthly_pulse = {
	events = {
		flavor_eth.103
	}
}
on_monarch_death = {
	if = {
		limit = { 
			has_country_modifier = EDG_ETH_help_of_Errani
		}
		remove_country_modifier = EDG_ETH_help_of_Errani
	}
	if = {
		limit = {
			has_government_mechanic = EDG_ETH_solomonic_authority_mechanic
		}
		add_government_power = {
			mechanic_type = EDG_ETH_solomonic_authority_mechanic
			power_type = EDG_ETH_solomonic_authority
			value = -10
		}
	}
}
on_queen_death = {
	if = {
		limit = { 
			has_consort_flag = EDG_ETH_Errani_flag
		}
		remove_country_modifier = EDG_ETH_help_of_Errani
	}
}
on_province_religion_converted = {
	if = {
		limit = {
			has_owner_religion = yes 
			owner = {
				has_government_mechanic = EDG_ETH_solomonic_authority_mechanic
			}
		}
		owner = {
			add_government_power = {
				mechanic_type = EDG_ETH_solomonic_authority_mechanic
				power_type = EDG_ETH_solomonic_authority
				value = 0.1
			}
		}
	}
}
on_core = {
	if = {
		limit = {
			from = {
				has_government_mechanic = EDG_ETH_solomonic_authority_mechanic
			}
		}
		from = {
			add_government_power = {
				mechanic_type = EDG_ETH_solomonic_authority_mechanic
				power_type = EDG_ETH_solomonic_authority
				value = 0.1
			}
		}
	}
}
