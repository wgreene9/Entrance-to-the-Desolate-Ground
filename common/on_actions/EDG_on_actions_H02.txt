
on_startup = {
	#dd适配点
	H02 = { country_event = { id = EDG_flavor_H02_mission.0 } }
}
on_monthly_pulse = {
	events = {
		EDG_flavor_H02_mission.1
		EDG_flavor_H02_mission.2
	}
}

on_new_monarch = {
	if = {
		limit = {
			has_country_flag = EDG_mission_on_new_monarch_provide_100_mil_power_flag
		}
		add_mil_power = 100
	}
}
on_general_recruited = {
	if = {
		limit = {
			has_country_flag = EDG_on_general_recruited_provide_5_legitimacy_flag
		}
		add_legitimacy = 5
	}
}
