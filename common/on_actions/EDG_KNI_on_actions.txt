on_startup = {
	if = {
		limit = {
			tag = KNI
		}
		country_event = {
			id = EDG_flavorKNI_mission.20
			days = 1
		}
	}
}

on_province_religion_converted = {
	on_province_religion_changed_effect = yes
	on_province_religion_converted_effect = yes
  if={
	limit={#传教加人力
	  owner = {
             has_country_flag=EDG_KNI_unlock_1_mil_flag
	  }	
	} 
	add_base_manpower = 1
  }
  if={
	limit={#传教加税收
	  owner = {
		has_country_flag=EDG_KNI_unlock_1_adm_flag
      }	
	} 
	add_base_tax = 1
  }
  if={
	limit={#传教加生产
	  owner = {
		has_country_flag=EDG_KNI_unlock_1_dip_flag
      }	
	} 
	add_base_production = 1
  }
  if={
	limit={#传教加全部
	  owner = {
		has_country_flag=EDG_KNI_unlock_1_all_flag
	  }
	} 
	add_base_production = 1
	add_base_manpower = 1
	add_base_tax = 1		
  }
}
