on_province_religion_changed = {
	if = {
		limit = {
			religion = catholic
		}
		if = {
			limit = {
				exists = PAP
			}
			PAP = {
				add_government_power = {
					mechanic_type = EDG_PAP_papal_authority_mechanic
					power_type = EDG_PAP_papal_authority
					value = 0.1
				}
			}
		}
	}
	if = {
		limit = {
			OR = {
				religion = EDG_Great_Celtic_Church
				religion = EDG_Waldenses
				religion = EDG_Lollards
				religion = EDG_Euchites
				religion = EDG_catharism
				religion = hussite
				religion = anglican
				religion = reformed
				religion = protestant
			}
		}
		if = {
			limit = {
				exists = PAP
			}
			PAP = {
				add_government_power = {
					mechanic_type = EDG_PAP_papal_authority_mechanic
					power_type = EDG_PAP_papal_authority
					value = -0.1
				}
			}
		}
	}
}

on_war_won = {
	if = {
		limit = {
			root = {
				religion = catholic
			}
			from = {
				OR = {
					religion = EDG_Great_Celtic_Church
					religion = EDG_Waldenses
					religion = EDG_Lollards
					religion = EDG_Euchites
					religion = EDG_catharism
					religion = hussite
					religion = anglican
					religion = reformed
					religion = protestant
					religion_group = muslim
				}
			}
		}
		if = {
			limit = {
				exists = PAP
			}
			PAP = {
				add_government_power = {
					mechanic_type = EDG_PAP_papal_authority_mechanic
					power_type = EDG_PAP_papal_authority
					value = 0.5
				}
			}
		}
	}
	if = {
		limit = {
			root = {
				OR = {
					religion = EDG_Great_Celtic_Church
					religion = EDG_Waldenses
					religion = EDG_Lollards
					religion = EDG_Euchites
					religion = EDG_catharism
					religion = hussite
					religion = anglican
					religion = reformed
					religion = protestant
					religion_group = muslim
				}
			}
			from = {
				religion = catholic
			}
		}
		if = {
			limit = {
				exists = PAP
			}
