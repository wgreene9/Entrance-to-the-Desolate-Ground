EDG_SWI_eden_dream_interactions_trigger = {
	if = {
		limit = { 
			has_country_modifier = EDG_SWI_$code$_country_modifier_1
		}
		has_government_power = {
			mechanic_type = EDG_SWI_eden_dream_mechanic
			power_type = EDG_SWI_eden_dream
			value = 15
		}
	}
	else_if = {
		limit = { 
			has_country_modifier = EDG_SWI_$code$_country_modifier_2
		}
		has_government_power = {
			mechanic_type = EDG_SWI_eden_dream_mechanic
			power_type = EDG_SWI_eden_dream
			value = 30
		}
	}
	else_if = {
		limit = { 
			has_country_modifier = EDG_SWI_$code$_country_modifier_3
		}
		has_government_power = {
			mechanic_type = EDG_SWI_eden_dream_mechanic
			power_type = EDG_SWI_eden_dream
			value = 45
		}
	}
	else_if = {
		limit = { 
			has_country_modifier = EDG_SWI_$code$_country_modifier_3
		}
		has_government_power = {
			mechanic_type = EDG_SWI_eden_dream_mechanic
			power_type = EDG_SWI_eden_dream
			value = 60
		}
	}
	else = {
		custom_trigger_tooltip = {
			tooltip = EDG_SWI_eden_dream_interactions_locks
			always = no
		}
	}
}
