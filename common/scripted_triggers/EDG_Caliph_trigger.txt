
EDG_caliph_trigger = {#是哈里发
	religion_group = muslim
	OR = {
		has_country_modifier = unified_islam
		has_reform = caliphate_reform
		has_reform = caliphate_theocracy_reform
		has_government_attribute = EDG_caliph_yes_ability
		has_country_modifier = EDG_Usurping_the_sunni_Caliphate_modifier
	}
}
EDG_Caliph_ability_trigger = {#可以发动哈里发的通用能力
	#吉哈德
	religion_group = muslim
	OR = {
		EDG_caliph_trigger = yes
		any_owned_province = {
			has_province_modifier = EDG_Abbasid_Caliphate_province_modifier
		}
		any_owned_province = {
			has_province_modifier = EDG_Fatima_Caliphate_province_modifier
		}
		# has_estate_privilege = estate_EDG_FYM_Abbasid_caliphate_privilege
		# has_estate_privilege = estate_EDG_FYM_Abbasid_caliphate_privilege_2
		# has_estate_privilege = estate_EDG_FYM_Fatima_caliphate_privilege
		# has_estate_privilege = estate_EDG_FYM_Fatima_caliphate_privilege_2
	}
}

#逊尼派
EDG_sunni_Caliph_ability_trigger = {#是或者控制了逊尼派哈里发
	#逊尼派哈里发制裁
	religion_group = muslim
	custom_trigger_tooltip = {
		tooltip = EDG_sunni_Caliph_moving_global_flag_trigger_tt
		NOT = { has_global_flag = EDG_sunni_Caliph_moving_global_flag }
	}
	custom_trigger_tooltip = {
		tooltip = EDG_sunni_Caliph_ability_trigger_tt
		OR = {
			AND = {
				tag = ABS
				religion = sunni
			}
			AND = {
				has_country_modifier = unified_islam
				religion = sunni
			}
			AND = {
				NOT = { exists = ABS }
				any_owned_province = {
					has_province_modifier = EDG_Abbasid_Caliphate_province_modifier
				}
			}
			has_country_modifier = EDG_Usurping_the_sunni_Caliphate_modifier
		}
	}
}
EDG_sunni_Caliph_decree_button_trigger = {#可以颁布逊尼派哈里发诏书
	check_variable = {
		which = EDG_sunni_Caliph_authority_var
		value = 20
	}
}
