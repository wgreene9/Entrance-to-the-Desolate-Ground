EDG_BAV_is_bavarian_country_trigger = {
	custom_trigger_tooltip = {
		tooltip = EDG_BAV_is_bavarian_country_trigger_tt
		OR = {
			is_or_was_tag = {
				tag = BAV
			}
			is_or_was_tag = {
				tag = LBV
			}
		}
	}
}

EDG_BAV_is_bavarian_province_trigger = {
	OR = {
		area = east_bavaria_area
		area = lower_bavaria_area
		area = upper_bavaria_area
	}
}

EDG_BAV_is_german_province_trigger = {
	OR = {
		region = north_german_region
		region = south_german_region
	}
}

EDG_BAV_is_heretic_province_of_owner_trigger = {
	custom_trigger_tooltip = {
		tooltip = EDG_BAV_is_heretic_province_of_owner_trigger_tt

			religion = owner
		}
	}
}

EDG_BAV_can_build_beer_factory_in_province_trigger = {
	trade_goods = grain
	is_state = yes
	is_state_core = owner
	NOT = {
		has_construction = any
	}
	EDG_BAV_not_has_beer_factory_modifier_trigger = yes
}

EDG_BAV_has_beer_factory_modifier_trigger = {
	custom_trigger_tooltip = {
		tooltip = EDG_BAV_has_beer_factory_modifier_trigger_tt
		OR = {
			has_province_modifier = EDG_BAV_beer_factory_modifier_1
			has_province_modifier = EDG_BAV_beer_factory_modifier_2
			has_province_modifier = EDG_BAV_beer_factory_modifier_3
			has_province_modifier = EDG_BAV_beer_factory_modifier_4
			has_province_modifier = EDG_BAV_beer_factory_modifier_5
			has_province_modifier = EDG_BAV_beer_factory_modifier_6
		}
	}
}

EDG_BAV_not_has_beer_factory_modifier_trigger = {
	custom_trigger_tooltip = {
		tooltip = EDG_BAV_not_has_beer_factory_modifier_trigger_tt
		EDG_BAV_has_beer_factory_modifier_trigger = no
	}
}

EDG_BAV_GUI_religious_congress_button_trigger = {
	religion = catholic
	if = {
		limit = {
			NOT = {
				has_government_attribute = EDG_BAV_establish_religious_agency_attribute
			}
		}
		papal_influence = 25
	}
	custom_trigger_tooltip = {
		tooltip = EDG_BAV_GUI_religious_congress_$religion$_button_trigger_tt_1
		has_country_flag = EDG_BAV_GUI_religious_congress_$religion$_button_flag
	}
	custom_trigger_tooltip = {
		tooltip = EDG_BAV_GUI_religious_congress_$religion$_button_trigger_tt_2
		NOT = {
			has_country_modifier = EDG_BAV_GUI_religious_congress_$religion$_button_1_modifier
		}
		NOT = {
			has_country_modifier = EDG_BAV_GUI_religious_congress_$religion$_button_2_modifier
		}
		NOT = {
			has_country_modifier = EDG_BAV_GUI_religious_congress_$religion$_button_3_modifier
		}
		NOT = {
			has_country_modifier = EDG_BAV_GUI_religious_congress_$religion$_button_4_modifier
		}
	}
}
