
EDG_PER_Shahanshah_GUI_trigger = {
	not = { has_country_flag = EDG_PER_Shahanshah_GUI_off_flag }
	EDG_is_iranian_tag_trigger = yes
	OR = {
		has_country_modifier = EDG_PER_Shahanshah_the_tag_modifier
		any_country = {
			has_country_modifier = EDG_PER_Shahanshah_the_tag_modifier
		}
	}
}
EDG_PER_Shahanshah_decree_enable_trigger = {
	custom_trigger_tooltip = {
		tooltip = EDG_PER_Shahanshah_decree_enable_trigger_tt
		has_country_modifier = EDG_PER_Shahanshah_the_tag_modifier
		OR = {
			AND = {
				has_government_mechanic = persian_influence_mechanic
				has_government_power = { mechanic_type = persian_influence_mechanic power_type = persian_influence value = 25 }
			}
			NOT = { has_country_modifier = EDG_PER_Shahanshah_decree_in_10_years_modifier }
		}
	}
}
EDG_PER_Shahanshah_the_reform_has_competed_trigger = {
	custom_trigger_tooltip = {
		tooltip = EDG_PER_Shahanshah_the_reform_has_competed_trigger_tt
	}
}
EDG_PER_Shahanshah_can_Promote_reform_trigger = {
	if = {
		limit = { EDG_PER_Shahanshah_trigger = yes }
		check_variable = {
			which = EDG_PER_Shahanshah_authority_var
			value = 100
		}
	}
	else = {
		EDG_PER_Shahanshah_trigger = yes
	}
}
EDG_PER_Shahanshah_trigger = {
	custom_trigger_tooltip = {
		tooltip = EDG_PER_Shahanshah_trigger_tt
		tag = event_target:EDG_PER_Shahanshah_target
		has_country_modifier = EDG_PER_Shahanshah_the_tag_modifier
	}
}
NOT_EDG_PER_Shahanshah_trigger = {
	custom_trigger_tooltip = {
		tooltip = NOT_EDG_PER_Shahanshah_trigger_tt
		NOT = { tag = event_target:EDG_PER_Shahanshah_target }
	}
}
