EDG_FOI_UCI_UI_select_trait_trigger = {
	if = {
		limit = { 
			NOT = {
				OR = {
					has_country_flag = FOI_UCI_UI_court_trait_select_button_1_flag
					has_country_flag = FOI_UCI_UI_court_trait_select_button_2_flag
					has_country_flag = FOI_UCI_UI_court_trait_select_button_3_flag
				}
			}
		}
		custom_trigger_tooltip = {
			tooltip = FOI_UCI_UI_please_select_a_slot
			OR = {
				has_country_flag = FOI_UCI_UI_court_trait_select_button_1_flag
				has_country_flag = FOI_UCI_UI_court_trait_select_button_2_flag
				has_country_flag = FOI_UCI_UI_court_trait_select_button_3_flag
			}
		}
	}
	if = {
		limit = { 
			OR = {
				has_country_flag = FOI_UCI_UI_court_trait_select_button_1_flag
				has_country_flag = FOI_UCI_UI_court_trait_select_button_2_flag
				has_country_flag = FOI_UCI_UI_court_trait_select_button_3_flag
			}
		}
		custom_trigger_tooltip = {
			tooltip = FOI_UCI_UI_has_select_this_trait
			if = {
				limit = { 
					has_country_flag = FOI_UCI_UI_court_trait_select_button_1_flag
				}
				NOT = {
					OR = {
						has_country_flag = FOI_UCI_UI_court_trait_button_$serial$_2_flag
						has_country_flag = FOI_UCI_UI_court_trait_button_$serial$_3_flag
					}
				}
			}
			else_if = {
				limit = { 
					has_country_flag = FOI_UCI_UI_court_trait_select_button_2_flag
				}
				NOT = {
					OR = {
						has_country_flag = FOI_UCI_UI_court_trait_button_$serial$_1_flag
						has_country_flag = FOI_UCI_UI_court_trait_button_$serial$_3_flag
					}
				}
			}
			else_if = {
				limit = { 
					has_country_flag = FOI_UCI_UI_court_trait_select_button_3_flag
				}
				NOT = {
					OR = {
						has_country_flag = FOI_UCI_UI_court_trait_button_$serial$_2_flag
						has_country_flag = FOI_UCI_UI_court_trait_button_$serial$_1_flag
					}
				}
			}
		}
	}
	[[use_empirical]
		check_variable = {
			which = EDG_FOI_UCI_empirical_value
			value = 200
		}
	]
}
