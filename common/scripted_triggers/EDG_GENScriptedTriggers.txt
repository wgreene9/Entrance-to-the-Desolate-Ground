		tooltip = EDG_GEN_main_surface_button_2.conditon.tt
		NOT = {
			has_country_flag = EDG_GEN_blocked_account_flag
		}
	}

		has_country_modifier = EDG_GEN_debt_default
	}

			superregion = europe_superregion

		is_bankrupt = no

			is_rival = root
		}

		}
	}
}
