EDG_BYZ_check_leisure_slot = {
	[[cost]
		owner = {
			treasury = $cost$
		}
	]
	if = {
		limit = { 
			check_variable = {
				which = EDG_BYZ_province_window_var
				value = 1
			}
			NOT = {
				check_variable = {
					which = EDG_BYZ_province_window_var
					value = 2
				}
			}
		}
		check_variable = {
			which = EDG_BYZ_constantinople_1_slot_leisure_var
			value = $value$
		}
	}
	else_if = {
		limit = { 
			check_variable = {
				which = EDG_BYZ_province_window_var
				value = 2
			}
			NOT = {
				check_variable = {
					which = EDG_BYZ_province_window_var
					value = 3
				}
			}
		}
		check_variable = {
			which = EDG_BYZ_constantinople_2_slot_leisure_var
			value = $value$
		}
	}
	else_if = {
		limit = { 
			check_variable = {
				which = EDG_BYZ_province_window_var
				value = 3
			}
			NOT = {
				check_variable = {
					which = EDG_BYZ_province_window_var
					value = 4
				}
			}
		}
		check_variable = {
			which = EDG_BYZ_constantinople_3_slot_leisure_var
			value = $value$
		}
	}
	else_if = {
		limit = { 
			check_variable = {
				which = EDG_BYZ_province_window_var
				value = 4
			}
			NOT = {
				check_variable = {
					which = EDG_BYZ_province_window_var
					value = 5
				}
			}
		}
		check_variable = {
			which = EDG_BYZ_constantinople_4_slot_leisure_var
			value = $value$
		}
	}
	else = {
		always = no
	}
}
