EDG_BOH_tarot_index_common_trigger = {
	if = {
		limit = { 
			has_country_flag = EDG_BOH_tarot_index_$id$_locked_perpetual_flag
		}
		custom_trigger_tooltip = {
			tooltip = EDG_BOH_tarot_index_locked_perpetual
			always = no
		}
	}
	if = {
		limit = { 
			has_country_modifier = EDG_BOH_tarot_modifier_index_$id$
		}
		NOT = {
			has_country_modifier = EDG_BOH_tarot_modifier_index_$id$
		}
	}
	if = {
		limit = { 
			has_country_flag = EDG_BOH_select_tarot_index_flag
			has_country_flag = EDG_BOH_tarot_index_$id$_locked_flag
		}
		custom_trigger_tooltip = {
			tooltip = EDG_BOH_tarot_index.locked.tt
			always = no
		}
	}
}
