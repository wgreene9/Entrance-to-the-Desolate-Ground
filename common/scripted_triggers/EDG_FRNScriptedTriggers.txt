EDG_FRN_delegation_decision_trigger = {

		has_country_flag = EDG_FRN_delegation_control_flag
	}
	$province$ = {
		NOT = {
			has_great_project = {
				type = $great_project$
				tier = 3
			}
		}
	}
}

EDG_FRN_exploration_of_building_materials_trigger = {

		template = merc_EDG_FRN_building_materials_exploration_team

			controlled_by = owner
			if = {
				limit = { 
					OR = {
						province_id = 225
						province_id = 410
						province_id = 361
						province_id = 38
						province_id = 97
						province_id = 183
						province_id = 118
						province_id = 134
						province_id = 266
						province_id = 12
						province_id = 257
						province_id = 280
						province_id = 294
						province_id = 151
						province_id = 1878
					}
				}
				NOT = {
					development = 75
				}
			}
			else = {
				NOT = {
					development = 30
				}
			}
			NOT = {
				owned_by = root
			}
			OR = {
				trade_goods = glass
				trade_goods = iron
				trade_goods = copper
				trade_goods = gold
				trade_goods = gems
				has_terrain = forest
				has_terrain = woods
				has_terrain = mountain
				has_terrain = hills
			}
		}
	}
}
