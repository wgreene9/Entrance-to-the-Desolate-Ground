	adm_power = 200
	dip_power = 200
	mil_power = 200
	NOT = {
		has_government_power = {
			mechanic_type = ottoman_decadence_mechanic
			power_type = ottoman_decadence
			value = 90
		}
	}
	has_government_power = {
		mechanic_type = EDG_TUR_Monarchy_ability_2
		power_type = EDG_TUR_Monarchy_mil_power
		value = 25
	}
	has_government_power = {
		mechanic_type = EDG_TUR_Monarchy_ability_3
		power_type = EDG_TUR_Monarchy_adm_power
		value = 25
	}
	not = { has_country_modifier = EDG_TUR_gaomen_decree_in_10_years_modifier }
}

EDG_TUR_gaomen_can_Promote_reform_trigger = {
	if = {
		limit = { EDG_TUR_gaomen_trigger = yes }
		NOT = {
			has_government_power = {
				mechanic_type = ottoman_decadence_mechanic
				power_type = ottoman_decadence
				value = 97.5
			}
		}
		has_government_power = {
			mechanic_type = EDG_TUR_Monarchy_ability_2
			power_type = EDG_TUR_Monarchy_mil_power
			value = 5
		}
		has_government_power = {
			mechanic_type = EDG_TUR_Monarchy_ability_3
			power_type = EDG_TUR_Monarchy_adm_power
			value = 5
		}
	}
	else = {
		EDG_TUR_gaomen_trigger = yes
	}
}

}
