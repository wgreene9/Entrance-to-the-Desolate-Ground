riga_first_mission_tree_common_trigger = {
	NOT = {
		OR = {
			has_country_flag = rig_second_mission_tree
			has_country_flag = rig_third_mission_tree
			has_country_flag = already_begin_riga_dawn_of_war
			has_country_flag = begin_battle_of_heaven_flag
		}
	}
}
riga_second_mission_tree_common_trigger = {
	has_country_flag = rig_second_mission_tree
	NOT = {
		OR = {
			has_country_flag = begin_battle_of_heaven_flag
			has_country_flag = rig_third_mission_tree
		}
	}
}
riga_third_mission_tree_common_trigger = {
	has_country_flag = rig_third_mission_tree
	NOT = {
		has_country_flag = begin_battle_of_heaven_flag
	}
}
riga_subject_of_subject_owned_land = {
	custom_trigger_tooltip = {
		tooltip = riga_subject_of_subject_owned_land.tt
		OR = {
			country_or_non_sovereign_subject_holds = $who$
			owner = {
				is_subject = yes
				NOT = {
					is_subject_of = $who$
				}
				overlord = {
					is_subject_of = $who$
				}
			}
		}
	}
}
