EDG_FCC_wine_lord = { #葡萄酒领主
	lock_level_when_selected = yes
	potential = {
		has_reform = EDG_FCC_wine_lord
	}
	trigger = {
		has_reform = EDG_FCC_wine_lord
	}
	icon = ""
	allow_normal_conversion = no
	fixed_rank = 1

	modifiers = {
		prestige_decay = -0.01
		legitimacy = 1
		diplomats = 1
	}
	effect = {
		custom_tooltip = EDG_FCC_wine_lord.effect.tt
		hidden_effect = {
			every_owned_province = {
				if = {
					limit = {
						trade_goods = wine
					}
					add_province_modifier = { 
						name = EDG_FCC_wine_extra_modifier
						duration = -1
						hidden = yes
					}
				}
			}
		}
	}
	removed_effect = {
		hidden_effect = {
			every_owned_province = {
				limit = {
					has_province_modifier = EDG_FCC_wine_extra_modifier
				}
				remove_province_modifier = EDG_FCC_wine_extra_modifier
			}
		}
	}
	custom_attributes = {
		no_crownland_money_loss_to_age = yes
		can_sell_crownland_to_single_estates = yes
		EDG_extra_income_for_estate_land = yes
		EDG_enable_grape_farm_unconditional = yes
	}
}
