
EDG_TUR_bektashi_holy_order = {
    icon = GFX_holy_order_bektashi
    color = { 57 160 101 }
    cost = 100
    cost_type = mil_power
    trigger = {
        #tag = MHM
        has_country_flag = EDG_TUR_bektashi_holy_order_enable_flag
    }

    per_province_abandon_effect = {
        add_base_manpower = -1
        hidden_effect = {
            if = {
                limit = { NOT = { base_manpower = 1 } }
                set_base_manpower = 1
            }
            remove_province_modifier = EDG_TUR_bektashi_holy_order_mark_province_modifier
        }
    }
    per_province_effect = {
        add_base_manpower = 1
        hidden_effect = {
            add_province_modifier = {
                name = EDG_TUR_bektashi_holy_order_mark_province_modifier
                duration = -1
                hidden = yes
            }
        }
    }

    modifier = {
        local_manpower = 0.5
        local_tolerance_of_heretics = 3
        local_tolerance_of_heathens = 3
        local_monthly_devastation = -0.1
    }

    ai_priority = {
        factor = 1
    }

    localization = EDG_TUR_bektashi_holy_order_localization
}
