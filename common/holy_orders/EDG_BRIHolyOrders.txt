
EDG_BRI_procuratorate_holy_order = {
    icon = GFX_holy_order_benedictines
    color = { 101 57 160  }
    cost = 50
    cost_type = adm_power
    trigger = {
        #tag = MHM
        has_country_flag = EDG_BRI_procuratorate_holy_order_enable_flag
    }

    per_province_abandon_effect = {
        add_base_tax = -1
        hidden_effect = {
            if = {
                limit = { NOT = { base_tax = 1 } }
                set_base_tax = 1
            }
            remove_province_modifier = EDG_BRI_procuratorate_holy_order_mark_province_modifier
        }
    }
    per_province_effect = {
        add_base_tax = 1
        hidden_effect = {
            add_province_modifier = {
                name = EDG_BRI_procuratorate_holy_order_mark_province_modifier
                duration = -1
                hidden = yes
            }
        }
    }

    modifier = {
        local_state_maintenance_modifier = -0.35
        local_governing_cost = -0.15
    }

    ai_priority = {
        factor = 1
    }

    localization = EDG_BRI_procuratorate_holy_order_localization
}
