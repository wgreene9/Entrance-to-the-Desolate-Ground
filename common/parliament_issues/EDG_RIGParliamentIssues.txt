rig_cultivate_act = {
	category = 9

	allow = {
		has_government_attribute = riga_parliament
		any_subject_country = {
			any_owned_province = {
				is_core = owner
				NOT = {
					development = 10
				}
			}
		}
	}
	effect = {
		custom_tooltip = rig_cultivate_act.effect.tt
		add_adm_power = 50
		add_dip_power = 50
		add_mil_power = 50
		hidden_effect = {
			random_province = {
				limit = {
					owner = {
						is_subject_of = root
					}
					is_core = owner
					NOT = {
						development = 10
					}
				}
				while = {
					limit = { 
						NOT = {
							development = 10
						}
					}
					add_random_development = 1
				}
			}
		}
	}
	modifier = {
		development_cost_modifier = -0.25
	}
	ai_will_do = {
		factor = 1
	}
}
