    alert_index = 35

        has_country_flag = Allow_strategic_trade_flag

			limit = {

			}
		}
	}
    on_accept = {
        if = {
            limit = {
                FROM = {
                    capital_scope = {
                        OR = {
                            superregion = europe_superregion
                           	superregion = eastern_europe_superregion
                        }
                    }
                }
            }
			FROM = {
				clear_global_event_target = EDG_ERS_E_Seller
				save_global_event_target_as = EDG_ERS_E_Seller
			}
            country_event = {
                id = mission_EDGERSmission.83
            }

        else_if = {
			limit = {
				FROM = {
					capital_scope = {
						OR = {
							superregion = china_superregion
							superregion = tartary_superregion

						}
					}

				id = mission_EDGERSmission.82

				which = ai_value
				value = -20
			}
		}
        add_entry = {
