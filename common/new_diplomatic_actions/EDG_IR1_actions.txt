EDG_IR1_subject_type_1_create = {#建立伊朗自治领地
	category = influence
	alert_index = 17 #doesn't matter. it is forced onto the subject
	alert_tooltip = sell_indulgence_alert_tooltip
	is_visible = {
		has_country_flag = EDG_IR1_overlord_1_flag 
		FROM = {
			vassal_of = root 
			or = {
				is_subject_of_type = vassal
				is_subject_of_type = client_vassal
			}
			
		}
	}
	is_allowed = {
		treasury = 100
		dip_power = 50
		custom_trigger_tooltip = {
			tooltip = EDG_FEO_trade_subject_type_create_1_tooltip
			NOT = {
				any_subject_country = {
					is_subject_of_type = EDG_IR1_subject_type_1
					capital_scope = {
						region_for_scope_province = {
							FROM = {
								capital_scope = {
									province_id = PREV_PREV
								}
							}
						}
					}
				}
			}
		}
	}
	on_accept = {
		add_dip_power = -50
		add_treasury = -100
		create_subject = {
			subject_type = EDG_IR1_subject_type_1
			subject = FROM
		}
		from = {
			change_government = republic
			add_government_reform = trade_company_government
		}
	}
}

EDG_IR1_subject_type_1_inherit = {#强行剥夺领地
	category = influence
	alert_index = 17 #doesn't matter. it is forced onto the subject
	alert_tooltip = sell_indulgence_alert_tooltip
	is_visible = {
		has_country_flag = EDG_IR1_overlord_1_flag 
		has_country_flag = EDG_IR1_reform_select_7_flag
		FROM = {
			is_subject_of_type = EDG_IR1_subject_type_1
		}
	}
	is_allowed = {
		stability = 1
		dip_power = 100
		prestige = 0
	}
	on_accept = {
		add_stability = -3
		add_dip_power = -100
		add_prestige = -15
		FROM = {
			every_province = {
				limit = {
					owned_by = FROM
				}
				cede_province = ROOT
			}
		}
	}
}
