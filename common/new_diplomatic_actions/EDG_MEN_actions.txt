EDG_MEN_turkey_culture_change = {
	#改变主流文化为土耳其
	category = influence
	is_visible = {
		has_country_flag = EDG_MEN_3_29_flag
		or = {
			tag = RUM
			was_tag = RUM
		}
		FROM = {
			is_subject_of = root
			NOT = {
				primary_culture = turkish
			}
		}
	}
	is_allowed = {
		if = {
			limit = {
				is_at_war = yes
			}
			custom_trigger_tooltip = {
				tooltip = ROOT_IS_AT_WAR_TT
				is_at_war = no
			}
		}
		if = {
			limit = {
				FROM = {
					is_at_war = yes
				}
			}
			custom_trigger_tooltip = {
				tooltip = FROM_IS_AT_WAR_TT
				FROM = {
					is_at_war = no
				}
			}
		}
		FROM = {
			is_subject_of = root
			NOT = {
				dominant_culture = turkish
			}
		}
	}
	on_accept = {
		from = {
			change_primary_culture = turkish
		}
	}
}
