EDG_H04_white_peace = {
	category = influence
	require_acceptance = no
	is_visible = {
		has_government_mechanic = EDG_H04_palace_splendor_mechanic
		has_country_flag = EDG_H04_enable_white_peace_action_flag
		is_subject = no
		from = {
			is_subject = no
		}
	}
	is_allowed = {
		from = {
			is_part_of_hre = yes
		}
		war_with = from
		has_government_power = {
			mechanic_type = EDG_H04_palace_splendor_mechanic
			power_type = EDG_H04_palace_splendor
			value = 25
		}
	}
	on_accept = {
		white_peace = from
		add_government_power = {
			mechanic_type = EDG_H04_palace_splendor_mechanic
			power_type = EDG_H04_palace_splendor
			value = -25
		}
	}
	ai_will_do = {
		always = no
	}
}
