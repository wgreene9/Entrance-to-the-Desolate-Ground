EDG_HSA_join_trading_city = {
	category = influence
	require_acceptance = yes
	alert_index = 35
	alert_tooltip = EDG_HSA_join_trading_city_alert_tooltip
	is_allowed = {
		from = {
			
			check_variable = {
				which = EDG_HSA_trade_city_count
				value = 1
			}
		}
	}
	is_visible = {
		ai = no
		NOT = {
			num_of_cities = 2
		}
		from = {
			has_government_attribute = enable_EDG_HSA_hanseatic_cities
		}
	}
	on_accept = {
		from = {
			EDG_HSA_create_vassal_effect = {
				who = root
			}
			subtract_variable = {
				which = EDG_HSA_trade_city_count
				value = 1
			}
		}
	}
	ai_will_do = {
		OR = {
			AND = {
				num_of_loans = 1
				is_in_deficit = yes
				is_subject = no
				is_at_war = no
			}
			AND = {
				is_in_trade_league_with = from
				from = {
					is_trade_league_leader = yes
				}
			}
			AND = {
				alliance_with = root
				has_opinion = {
					who = from
					value = 100
				}
			}
		}
		capital_scope = {
			province_trade_power = 10
		}
	}
	ai_acceptance = {
		add_entry = {
			name = BASE_REJECTION
			limit = {
				always = yes
			}
			export_to_variable = {
				variable_name = ai_value
				value = 1
			}
			multiply_variable = {
				which = ai_value
				value = -25
			}
		}
        add_entry = {
            name = DIPLO_OPPINION
			export_to_variable = {
				variable_name = ai_value
				value = opinion
				who = FROM
				with = ROOT
			}
			multiply_variable = {
				which = ai_value 
				value = 0.2
			}
        }
		add_entry = {
			name = AI_TRUST_TOWARDS_ROOT
			export_to_variable = {
				variable_name = ai_value
				value = trust
				who = FROM
				with = ROOT
			}			
			subtract_variable = { 
				which = ai_value
				value = 50
			}
			#1 trust = +4 ai points
			multiply_variable = {
				which = ai_value
				value = 0.1
			}
		}
		add_entry = {
			name = AI_DIPLOMATIC_REPUTATION
			export_to_variable = {
				variable_name = ai_value
				value = trigger_value:diplomatic_reputation
			}
			#1 rep = 4
			multiply_variable = {
				which = ai_value
				value = 4
			}
		}
		add_entry = {
			name = attitude_neutral_DIPLO_ATTITUDE_TT
			limit = { FROM = { ai_attitude = { who = ROOT attitude = attitude_neutral } } }
			export_to_variable = { variable_name = ai_value value = 1 }
			multiply_variable = { which = ai_value value = -20 }
		}
		add_entry = {
			name = attitude_rivalry_DIPLO_ATTITUDE_TT
			limit = { FROM = { ai_attitude = { who = ROOT attitude = attitude_rivalry } } }
			export_to_variable = { variable_name = ai_value value = 1 }
			multiply_variable = { which = ai_value value = -1000 }
		}
		add_entry = {
			name = attitude_hostile_DIPLO_ATTITUDE_TT
			limit = { FROM = { ai_attitude = { who = ROOT attitude = attitude_hostile } } }
			export_to_variable = { variable_name = ai_value value = 1 }
			multiply_variable = { which = ai_value value = -1000 }
		}
		add_entry = {
			name = attitude_friendly_DIPLO_ATTITUDE_TT
			limit = { FROM = { ai_attitude = { who = ROOT attitude = attitude_friendly } } }
			export_to_variable = { variable_name = ai_value value = 1 }
			multiply_variable = { which = ai_value value = 10 }
		}
		add_entry = {
			name = attitude_threatened_DIPLO_ATTITUDE_TT
			limit = { FROM = { ai_attitude = { who = ROOT attitude = attitude_threatened } } }
			export_to_variable = { variable_name = ai_value value = 1 }
			multiply_variable = { which = ai_value value = 10 }
		}
		add_entry = {
			name = attitude_outraged_DIPLO_ATTITUDE_TT
			limit = { FROM = { ai_attitude = { who = ROOT attitude = attitude_outraged } } }
			export_to_variable = { variable_name = ai_value value = 1 }
			multiply_variable = { which = ai_value value = -1000 }
		}
		add_entry = {
			name = attitude_domineering_DIPLO_ATTITUDE_TT
			limit = { FROM = { ai_attitude = { who = ROOT attitude = attitude_domineering } } }
			export_to_variable = { variable_name = ai_value value = 1 }
			multiply_variable = { which = ai_value value = -1000 }
		}
	}
}

EDG_HSA_change_to_vassal = {
	category = influence
	require_acceptance = no
	is_visible = {
		from = {
			is_subject_of = root
			is_subject_of_type = EDG_HSA_trade_city
		}
		ai = no
	}
	is_allowed = {
		is_at_war = no
	}
	on_accept = {
		create_subject = {
			subject_type = vassal
			subject = from
		}
	}
	ai_will_do = {
		NOT = {
			tag = HSA
		}
	}
}
