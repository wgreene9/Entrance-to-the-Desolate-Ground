custom_button = {
	name = EDG_sunni_Caliph_decree_button_1
	tooltip = EDG_Caliph_decree_tooltip_1_tt
	potential = {  EDG_Caliph_GUI_trigger = yes }
	trigger = {
		if = {
			limit = { EDG_sunni_Caliph_ability_trigger = yes }
			EDG_sunni_Caliph_decree_button_trigger = yes
			if = { limit = { religion = sunni } NOT = { has_country_modifier = EDG_Caliph_sunni_decree_button_1_modifier_a } }
			else = { NOT = { has_country_modifier = EDG_Caliph_sunni_decree_button_1_modifier_b } }
		}
		else = { EDG_sunni_Caliph_ability_trigger = yes }	
	}
	effect = {
		EDG_sunni_Caliph_decree_cost_effect = yes
		EDG_sunni_Caliph_decree_button_1_scriped_effect = yes
	}
}
custom_icon = {
	name = EDG_sunni_Caliph_decree_button_1_check
	potential = {
		EDG_Caliph_GUI_trigger = yes
		OR = { 	has_country_modifier = EDG_Caliph_sunni_decree_button_1_modifier_a
