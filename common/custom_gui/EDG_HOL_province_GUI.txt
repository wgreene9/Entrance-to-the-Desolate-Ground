custom_button = {
	name = EDG_HOL_barycenter_button_1
	tooltip = EDG_HOL_barycenter_button_1.tt
	potential = {
	}
	trigger = {
		not = { development = 50 }
		OR = {
			owned_by = from
			owner = {
				is_subject_of = from
			}
		}
		from = {
			not = {
				num_of_loans = 1
			}
			treasury = 0
		}
	}
	effect = {
		add_base_tax = 2
		add_base_manpower = -1
		add_base_production = -1
		from = {
			add_years_of_income = -0.5
			add_adm_power = -50
			add_government_power = {
				mechanic_type = EDG_HOL_speculate_ability
				power_type = EDG_HOL_speculate_power
				value = 2
			}
		}
		hidden_effect = {
			change_variable = {
				which = EDG_HOL_BUTTON_VALUE_1
				value = 1
			}
		}
	}
}
