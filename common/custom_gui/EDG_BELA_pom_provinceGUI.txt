custom_button = { #领地发展计划：改善税收
	name = EDG_BELA_POM_GUI_province_ability_adm
	tooltip = EDG_BELA_POM_GUI_province_ability_adm_tooltip
	potential = {
		from = {
			has_country_flag = EDG_BELA_POM_WOL_4_2_flag
		}
	}
	trigger = {
		controlled_by = from
		not = {
			development = 25
		}
		from = {
			check_variable = {
				which = EDG_BELA_POM_develop_var
				value = 100
			}
		}
	}
	effect = { 
		if = {
			limit = {
				has_tax_building_trigger = yes
			}
			add_scaled_local_adm_power = 3
		}
		add_base_tax = 1
		from = {
			change_variable = {
				which = EDG_BELA_POM_develop_var
				value = -100
			}
			if = {
				limit = {
					has_government_mechanic = EDG_POM_germanization_mechanic
				}
				add_government_power = {
					mechanic_type = EDG_POM_germanization_mechanic
					power_type = EDG_POM_germanization_power
					value = -1
				}
			}
		}
	}
}
