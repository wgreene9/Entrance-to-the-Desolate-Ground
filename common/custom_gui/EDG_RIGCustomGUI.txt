	tooltip = rig_assisted_core_for_subject_tooltip

		from = {
			has_country_flag = rig_riga_conference_flag
		}
		owner = {
			OR = {
				is_subject_of = from
				AND = {
					NOT = {
						is_subject_of = from

						is_subject_of = from
					}
				}
			}
		}
		NOT = {
			is_core = owner
		}
	}

		}

			}
		}
	}

		custom_tooltip = rig_assisted_core_for_subject_effect_tooltip

				which = EDG_FOI_UCI_province_development

				limit = {

						which = EDG_FOI_UCI_province_development
						value = 1
					}
				}
				subtract_variable = {
					which = EDG_FOI_UCI_province_development
					value = 1
				}
				from = {
					add_adm_power = -5
				}
			}
			#add_core_construction = yes
			add_core = owner
		}
	}
