	name = EDG_WAL_main_button
	tooltip = EDG_WAL_main_button.tt
	potential = {
		from = {
			has_government_attribute = EDG_enable_WAL_crucifixion
		}
		owned_by = from
		NOT = {
			has_province_modifier = EDG_WAL_crucifixion_modifier
		}
	}
	trigger = {
		owned_by = from
	}
	effect = {
		owner = {
			change_variable = {
				which = EDG_WAL_main_button_use_var
				value = 1
			}
		}
		custom_tooltip = EDG_SPACE_TT
		from = {
			add_government_power = {
				mechanic_type = EDG_WAL_atrocity_mechanic
				power_type = EDG_WAL_atrocity
				value = 1
			}
		}
		if = {
			limit = { 
				from = {
					has_country_flag = EDG_WAL_mission_2_13_flag
				}
			}
			custom_tooltip = EDG_SPACE_TT
			add_province_modifier = {
				name = EDG_WAL_crucifixion_modifier
				duration = 1825
			}
		}
		else = {
			custom_tooltip = EDG_SPACE_TT
			add_province_modifier = {
				name = EDG_WAL_crucifixion_modifier
				duration = -1
			}
		}
		if = {
			limit = { 
				from = {
					has_country_flag = EDG_WAL_mission_5_9_flag
				}
			}
			custom_tooltip = EDG_SPACE_TT
			add_nationalism = -5
			
		}
		if = {
			limit = { 
				from = {
					has_country_flag = EDG_H06_enable_change_culture_modifier_for_penalty_flag
				}
			}
			custom_tooltip = EDG_SPACE_TT
			add_province_modifier = {
				name = EDG_H06_assimilate_culture
				duration = -1
			}
		}
	}
