

EDG_catharism_aspect_1 = {
	sprite = "GFX_dissolve_monasteries_icon"
	cost = 100
	trigger = {
		has_estate = estate_church
	}
	effect = {
		if = {
			limit = {
				has_country_modifier = EDG_catharism_aspect_1_modifier
			}
			custom_tooltip = EDG_catharism_aspect_1_tt_ing
		}
		else = {
			custom_tooltip = EDG_catharism_aspect_1_tt
		}
	  hidden_effect = {
		EDG_add_or_extend_country_modifier_effect = { MODIFIER = EDG_catharism_aspect_1_modifier DURATION = 5475 }
		change_estate_land_share = {
		   estate = estate_church
		   share = -3
	    }
	  }
	}
	ai_will_do = {
		factor = 1
		modifier = {
			factor = 2
			is_in_large_debt = yes
		}
	}
}
