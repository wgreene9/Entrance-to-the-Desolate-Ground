			}

				name = EDG_estate_Interracial_Expand_autonomous_communities_province_modifier

			}
		}
		else_if = {

			remove_province_modifier = EDG_estate_Interracial_Expand_autonomous_communities_province_modifier

				name = EDG_estate_Interracial_Expand_autonomous_communities_province_modifier
				duration = -1
			}
		}

			remove_province_modifier = EDG_estate_Interracial_Expand_autonomous_communities_province_modifier
