estate_dhimmi_secret_sacrifice = {#隐秘的圣火祭祀
	icon = privilege_grant_autonomy
	max_absolutism = -5
	conditional_modifier = { trigger = { has_absolutism_reduction_for_estate_privileges = yes } modifier = { max_absolutism = 1 } }
	influence = 0.1
    loyalty = 0.1
	is_valid = {
		OR = {
			tag = TAB
			was_tag = TAB
		}
		any_owned_province = {
			OR = {
				religion = sunni
				religion = zoroastrian
			}
		}
	}
	on_granted = {
		tooltip = {
			add_province_modifier = {
				name = new_TAB_buff_514
				duration = -1
			}
		}
		hidden_effect = {
			set_country_flag = edg_TAB_secret_sacrifice_flag
		}
	}
	on_granted_province = {
		if = {
			limit = {
				OR = {
					religion = sunni
					religion = zoroastrian
				}
			}
			add_province_modifier = {
				name = new_TAB_buff_514
				duration = -1
			}
		}
		else_if = {
			limit = {
				has_province_modifier = new_TAB_buff_514
			}
			remove_province_modifier = new_TAB_buff_514
		}
	}
	on_revoked = {
		custom_tooltip = new_TAB_buff_514_revoke_effect_tooltip
		hidden_effect = {
			clr_country_flag = edg_TAB_secret_sacrifice_flag
		}
	}
	on_revoked_province = {
		if = {
			limit = {
				has_province_modifier = new_TAB_buff_514
			}
			remove_province_modifier = new_TAB_buff_514
		}
	}
	on_invalid = {
		custom_tooltip = new_TAB_buff_514_revoke_effect_tooltip
		hidden_effect = {
			clr_country_flag = edg_TAB_secret_sacrifice_flag
		}
	}
	on_invalid_province = {
		if = {
			limit = {
				has_province_modifier = new_TAB_buff_514
			}
			remove_province_modifier = new_TAB_buff_514
		}
	}
	can_revoke = {
		always = no
	}
	penalties = {
		monthly_piety = -0.01
		tolerance_heretic = 6
		tolerance_of_heretics_capacity = 3
		tolerance_heathen = 6
		tolerance_of_heathens_capacity = 3
	}
	benefits = {
	}
}
