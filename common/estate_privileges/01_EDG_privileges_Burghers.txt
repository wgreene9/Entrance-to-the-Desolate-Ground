	max_absolutism = -5
	conditional_modifier = { trigger = { has_absolutism_reduction_for_estate_privileges = yes } modifier = { max_absolutism = 1 } }
	loyalty = 0.05
	influence = 0.05
	is_valid = {
		has_country_flag = estate_EDG_Alawite_tolerance_privilege_enable_flag
		religion_group = muslim
		NOT = { religion = EDG_Alawite }
		any_owned_province = {
			religion = EDG_Alawite
		}
		NOT = { overlord = { tag = RIG } }
	}
	can_select = {
		religion_group = muslim
		NOT = { religion = EDG_Alawite }
		any_owned_province = {
			religion = EDG_Alawite
		}
		NOT = { overlord = { tag = RIG } }
	}
	on_granted = {
		if = {
			limit = { has_estate_privilege = estate_church_anti_heresy_act }
			remove_estate_privilege = estate_church_anti_heresy_act
			add_estate_loyalty = {
				estate = estate_church
				loyalty = -20
			}
		}
		custom_tooltip = estate_EDG_Alawite_tolerance_privilege_effect_tooltip
		hidden_effect = {
			set_country_flag = EDG_Alawite_tolerance_privilege_orthodoxy
		}
	}
	on_granted_province = {
		if = {
			limit = {
				religion = EDG_Alawite
			}
			add_province_modifier = {
				name = EDG_Alawite_tolerance_guaranteed_autonomy_mod
				duration = -1
			}
		}
		else_if = {
			limit = {
				has_province_modifier = EDG_Alawite_tolerance_guaranteed_autonomy_mod
			}
			remove_province_modifier = EDG_Alawite_tolerance_guaranteed_autonomy_mod
		}
	}
	on_revoked = {
		custom_tooltip = estate_EDG_Alawite_tolerance_privilege_revoke_effect_tooltip
		hidden_effect = {
			clr_country_flag = EDG_Alawite_tolerance_privilege_orthodoxy
		}
	}
	on_revoked_province = {
		if = {
			limit = {
				has_province_modifier = EDG_Alawite_tolerance_guaranteed_autonomy_mod
			}
			remove_province_modifier = EDG_Alawite_tolerance_guaranteed_autonomy_mod
		}
	}
	on_invalid = {
		custom_tooltip = estate_EDG_Alawite_tolerance_privilege_revoke_effect_tooltip
		hidden_effect = {
			clr_country_flag = EDG_Alawite_tolerance_privilege_orthodoxy
		}
	}
	on_invalid_province = {
		if = {
			limit = {
				has_province_modifier = EDG_Alawite_tolerance_guaranteed_autonomy_mod
			}
			remove_province_modifier = EDG_Alawite_tolerance_guaranteed_autonomy_mod
		}
	}
	penalties = {
		church_loyalty_modifier = -0.1
	}
	benefits = {
	}
	ai_will_do = {
		factor = 1000
		modifier = {
			factor = 0
			NOT = {
				any_owned_province = {
					religion = EDG_Alawite
				}
			}
		}
		modifier = {
			factor = 0
			government = theocracy
		}
		modifier = {
			factor = 0
			is_subject = yes
			overlord = {
				religion = ROOT
				has_idea_group = religious_ideas
			}
		}
		modifier = {
			factor = 0
			is_subject = yes
			overlord = {
				tag = RIG
			}
		}
	}
