estate_EDG_BDX_greek_buddhist_art = {
	icon = privilege_build_jain_temple
	loyalty = 0.05
	influence = 0.1
	land_share = 0
	max_absolutism = 0

	is_valid = {
		has_country_flag = EDG_BDX_mission_greek_buddhist_art_flag
		religion = mahayana
	}

	benefits = {
		church_privilege_slots = 1
	}

	loyalty_scaled_conditional_modifier = {
		trigger = {
			always = yes
		}
		modifier = {
			tolerance_own = 2
			monthly_splendor = 4
		}
	}

	on_granted = {
		custom_tooltip = estate_EDG_BDX_greek_buddhist_art_tt1
		tooltip = {
			add_province_modifier = {
				name = EDG_BDX_greek_buddhist_art_province_modifier
				duration = -1
				desc = UNTIL_REVOKED
			}
		}
	}

	on_revoked = {
		custom_tooltip = estate_EDG_BDX_greek_buddhist_art_tt2
	}

	on_granted_province = {
		if = {
			limit = {
				religion = mahayana
				has_tax_building_trigger = yes
			}
			add_province_modifier = {
				name = EDG_BDX_greek_buddhist_art_province_modifier
				duration = -1
				desc = UNTIL_REVOKED
			}
		}
	}

	on_invalid_province = {
		if = {
			limit = { has_province_modifier = EDG_BDX_greek_buddhist_art_province_modifier }
			remove_province_modifier = EDG_BDX_greek_buddhist_art_province_modifier
		}
	}

	on_revoked_province = {
		if = {
			limit = { has_province_modifier = EDG_BDX_greek_buddhist_art_province_modifier }
			remove_province_modifier = EDG_BDX_greek_buddhist_art_province_modifier
		}
	}

	ai_will_do = {
		factor = 100
	}
}
